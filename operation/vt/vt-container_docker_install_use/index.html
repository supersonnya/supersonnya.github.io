Docker安装及使用

Aug 09, 2020

Reading time: 2 minutes.

安装Docker

# install docker
yum insatall docker-io
# start docker
systemctl start docker
# check if installed successfully
ps -ef|grep docker
# restart docker
systemctl restart docker
# stop docker
systemctl stop docker

Docker安装Mysql

# install mysql in docker
## search mysql
docker search mysql
## instal mysql
docker pull mysql
## check images
docker images
## build directory
cd /opt/
mkdir mysql_docker
cd mysql_docker/
echo $PWD
# start mysql
docker run --name mysqlserver -v $PWD/conf:/etc/mysql/conf.d -v $PWD/logs:/logs -v $PWD/data:/var/lib/mysql -e MYSQL_ROOT_PASSWORD=123456 -d -i -p 3306:3306 mysql:latest
# check start
cd /var/lib/docker/containers/
# watch mysql service
docker ps -a
# login mysql
docker exec -it mysqlserver bash
mysql -uroot -p
# config mysql, change password model
# 0.0.0.0 connect
use mysql;
select host, user from user;
ALTER USER 'root'@'%' IDENTIFIED WITH mysql_native_password BY '123456';
flush privileges;
# watch docker logs
docker logs -f --tail 10 container_id
# stop mysql
docker stop mysqlserver
# restart mysql
docker restart container_id

Docker安装Oracle

# search images
docker search oracle
# chose image
docker pull jaspeen/oracle-11g
# view image
dokcer images
# download intall files
linux.x64_11gR2_database_1of2.zip & linux.x64_11gR2_database_2of2.zip
# decompression files, chose a directory will be used later
/home/user/data/database
# install oracle
# privileged 给容器特权
# name 命名 p 映射端口 v 安装文件挂载目录
docker run --privileged --name oracle11g -p 1521:1521 -v /home/user/data:/install jaspeen/oracle-11g
# view run status
docker ps -a
# config
docker exec -it oracle11g /bin/bash
su - oracle
sqlplus / as sysdba
alter user scott accout unlock;
commit;
conn scott/tiger
host: localhost
post: 1521
sid: orcl
user: scott/tiger
password: 
# import dmp fils
create user <用户名> identified by sys default tablespace <表空间名>;
grant DBA to <用户名> with admin option;
create or replace directory dump_dir as '/usr/oracle/dump/dir';
grant read,write on directory dump_dir to user ;
# container run
mkdir /opt/oracle/dpdump
chown -R oracle:oinstall /opt/oracle/dpdump
su - oracle
# host run
docker cp demo.dmp oracle11g: /opt/oracle/dpdump/dir
